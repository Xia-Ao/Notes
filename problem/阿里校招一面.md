```js

    };
}

```
用闭包实现输出对应的4

```js
var a = [];
for (var i = 0; i < 10; i++) {
    a[i] = function () {
        var c = i;
        return function () {
            console.log(c)
        }
    }();
}
a[4]();  //4
```
